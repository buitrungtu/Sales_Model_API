use Seles_Model;
go

CREATE TABLE category
(
	category_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	category_code varchar(10),
	parent_id UNIQUEIDENTIFIER,
	title nvarchar(255),
	metaTitle nvarchar(255),
	slug nvarchar(100),
	content nvarchar(max)
)
go
CREATE TABLE product_category
(
	product_id UNIQUEIDENTIFIER,
	category_id UNIQUEIDENTIFIER
)
go
CREATE TABLE tag
(
	tag_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	tag_code varchar(10),
	title nvarchar(255),
	metaTitle nvarchar(255),
	slug nvarchar(100),
	content nvarchar(max)
)
go
CREATE TABLE product
(
	product_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	account_id UNIQUEIDENTIFIER,
	product_code varchar(100),
	title nvarchar(255),
	metaTitle nvarchar(255),
	slug nvarchar(100),
	summary nvarchar(100),
	type int,
	sku nvarchar(100),
	price float,
	discount float,
	quantity int,
	shop int,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE(),
	published_date datetime DEFAULT GETDATE(),
	start_date datetime DEFAULT GETDATE(),
	end_date datetime DEFAULT GETDATE()
)
go

CREATE TABLE product_meta
(
	Id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	product_id UNIQUEIDENTIFIER,
	key_meta nvarchar(50),
	url nvarchar(255),
	content nvarchar(max)
)
go
CREATE TABLE product_tag
(
	tag_id UNIQUEIDENTIFIER,
	product_id UNIQUEIDENTIFIER
)
go

CREATE TABLE product_review
(
	Id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	product_id UNIQUEIDENTIFIER,
	parent_id UNIQUEIDENTIFIER,
	title nvarchar(255),
	rating int,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	published_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE account_info
(
	Id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	account_id UNIQUEIDENTIFIER,
	firstName nvarchar(50),
	midName nvarchar(50),
	lastName nvarchar(50),
	mobile varchar(15),
	email varchar(255),
	line1 nvarchar(50),
	line2 nvarchar(50),
	city nvarchar(50),
	province nvarchar(50),
	country nvarchar(50),
	profile nvarchar(max),
	intro nvarchar(255),
	create_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE account
(
	account_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	avatar nvarchar(255),
	username nvarchar(50),
	password nvarchar(max),
	status int,
	address varchar(255),
	dob datetime,
	first_name nvarchar(100),
	last_name nvarchar(100),
	mobile varchar(15),
	last_login datetime DEFAULT GETDATE(),
	last_action datetime DEFAULT GETDATE(),
	last_ip varchar(255),
	email_backup varchar(255),
	display_name nvarchar(255),
	is_interested_account int,
	create_date datetime DEFAULT GETDATE()
)
go

CREATE TABLE role
(
	role_id INT PRIMARY KEY IDENTITY,
	name nvarchar(255)
)
go

CREATE TABLE account_role
(
	account_id UNIQUEIDENTIFIER,
	role_id int
)
go

CREATE TABLE orders
(
	orders_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	account_id UNIQUEIDENTIFIER,
	session_id varchar(100),
	token varchar(100),
	status int,
	subTotal float,
	itemDiscount float,
	tax float,
	shipping float,
	total float,
	promo varchar(50),
	discount float,
	grandTotal float,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE transactions
(
	transaction_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	account_id UNIQUEIDENTIFIER,
	order_id UNIQUEIDENTIFIER,
	code varchar(100),
	type int,
	mode int,
	status int,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE orders_item
(
	Id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	product_id UNIQUEIDENTIFIER,
	order_id UNIQUEIDENTIFIER,
	sku varchar(100),
	price float,
	discount float,
	quantity int,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE cart
(
	cart_id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	account_id UNIQUEIDENTIFIER,
	session_id varchar(100),
	token varchar(100),
	status int,
	content nvarchar(max),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE()
)
go
CREATE TABLE cart_item
(
	Id UNIQUEIDENTIFIER PRIMARY KEY default NEWID(),
	product_id UNIQUEIDENTIFIER,
	cart_id UNIQUEIDENTIFIER,
	account_id UNIQUEIDENTIFIER,
	order_id UNIQUEIDENTIFIER,
	sku varchar(100),
	price float,
	discount float,
	quantity int,
	active int,
	content nvarchar(max),
	code varchar(100),
	create_date datetime DEFAULT GETDATE(),
	update_date datetime DEFAULT GETDATE()
)
go
